Control Statements

cfif/cfelseif/cfelse

In tags... 
		<cfset myVar = 5 />
		<cfif myVar == 0>
			<!--- do something --->
		<cfelseif myVar == 5>
			<!--- do something different --->
		<cfelse>
			<!--- default case --->
		</cfif>
In cfscript... 
		myVar = 5;
		if( myVar == 0 ) {
			// do something
		} else if( myVar == 5 ) {
			// do something else
		} else {
			// default case
		}

cfswitch

In tags... 
		<cfset myVar = 5 />
		<cfswitch expression="#myVar#">
			<cfcase value="0">
				<!--- do something --->
			</cfcase>
			<cfcase value="5">
				<!--- do something different --->
			</cfcase>
			<cfdefaultcase>
				<!--- default case --->
			</cfdefaultcase>
		</cfswitch>
In cfscript... 
		myVar = 5;
		switch( myVar ) {
			case "0": 
				// do something
				break;
			case "5":
				// do something different
				break;
			default:
				// default case
		}
